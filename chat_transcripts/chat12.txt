# portfolio/order_types.py
from __future__ import annotations
from dataclasses import dataclass, field
from enum import Enum, auto
from typing import Optional, Dict, Any
from datetime import datetime


class Side(str, Enum):
    BUY = "BUY"
    SELL = "SELL"
    SELL_SHORT = "SELL_SHORT"
    BUY_TO_COVER = "BUY_TO_COVER"


class OrderType(str, Enum):
    MARKET = "MARKET"
    LIMIT = "LIMIT"
    STOP = "STOP"
    STOP_LIMIT = "STOP_LIMIT"


class TimeInForce(str, Enum):
    DAY = "DAY"
    GTC = "GTC"
    IOC = "IOC"     # Immediate-or-cancel
    FOK = "FOK"     # Fill or kill (simulated as IOC with all-or-none)


@dataclass
class Order:
    ts: datetime
    symbol: str
    side: Side
    qty: float
    type: OrderType = OrderType.MARKET
    tif: TimeInForce = TimeInForce.DAY
    limit_price: Optional[float] = None
    stop_price: Optional[float] = None
    client_order_id: Optional[str] = None
    edge_id: Optional[str] = None
    edge: Optional[str] = None
    category: Optional[str] = None
    meta: Dict[str, Any] = field(default_factory=dict)


@dataclass
class Fill:
    ts: datetime
    symbol: str
    side: Side
    qty: float
    fill_price: float
    commission: float = 0.0
    pnl: Optional[float] = None
    client_order_id: Optional[str] = None
    edge_id: Optional[str] = None
    edge: Optional[str] = None
    category: Optional[str] = None
    meta: Dict[str, Any] = field(default_factory=dict)

# portfolio/oms.py
from __future__ import annotations
from typing import List, Dict, Optional, Tuple
from dataclasses import dataclass, field
from pathlib import Path
from datetime import datetime
import csv

import numpy as np
import pandas as pd

from .order_types import Order, Fill, OrderType, TimeInForce, Side


@dataclass
class ExecutionParams:
    fee_per_share: float = 0.0
    slippage_bps: float = 0.0      # e.g., 5 = 5 bps
    borrow_fee_bps_daily: float = 0.0  # short borrow (ignored if not short)


@dataclass
class OMSState:
    cash: float = 100_000.0
    positions: Dict[str, float] = field(default_factory=dict)      # symbol -> qty
    avg_price: Dict[str, float] = field(default_factory=dict)      # symbol -> avg cost


class OMS:
    """
    Backtest-grade OMS that can be swapped with a live broker adapter later.
    - Supports MARKET, LIMIT, STOP, STOP_LIMIT
    - TIF: DAY, GTC, IOC, FOK
    - Simulates partials for IOC/FOK
    - Applies simple slippage model on fills
    """
    def __init__(self, exec_params: Optional[ExecutionParams] = None,
                 trades_path: str | Path = "data/trade_logs/trades.csv") -> None:
        self.exec = exec_params or ExecutionParams()
        self.state = OMSState()
        self.trades_path = Path(trades_path)
        self.trades_path.parent.mkdir(parents=True, exist_ok=True)
        if not self.trades_path.exists():
            with self.trades_path.open("w", newline="") as f:
                csv.writer(f).writerow([
                    "timestamp","ticker","side","qty","fill_price","commission","pnl",
                    "edge","edge_group","trigger","meta","edge_id","edge_category"
                ])

        # GTC order book keyed by symbol
        self._gtc_orders: Dict[str, List[Order]] = {}

    # ---------- User API ---------- #
    def submit(self, order: Order) -> None:
        # GTC orders persist in book; DAY orders are evaluated at submit bar
        book = self._gtc_orders.setdefault(order.symbol, [])
        if order.tif == TimeInForce.GTC:
            book.append(order)
        else:
            book.append(order)  # keep it simple: we’ll clear DAY orders after evaluation

    def cancel_symbol(self, symbol: str) -> None:
        self._gtc_orders[symbol] = []

    def get_positions(self) -> Dict[str, float]:
        return dict(self.state.positions)

    # ---------- Backtest step ---------- #
    def on_bar(self, ts: datetime, bar: pd.Series) -> List[Fill]:
        """
        Evaluate resting orders against the new bar (OHLC). Return realized fills.
        `bar` must include: Open, High, Low, Close
        """
        fills: List[Fill] = []
        symbol = bar.name if isinstance(bar.name, str) else bar.get("symbol", "")
        o, h, l, c = [float(bar[k]) for k in ("Open","High","Low","Close")]

        orders = self._gtc_orders.get(symbol, [])
        remaining: List[Order] = []

        for ord in orders:
            filled, partial, fill_list = self._simulate_order(ts, ord, o, h, l, c)
            fills.extend(fill_list)
            if not filled:
                # IOC/FOK flush if not fully filled; otherwise keep
                if ord.tif in (TimeInForce.IOC, TimeInForce.FOK):
                    continue
                remaining.append(ord)

        # keep remaining
        self._gtc_orders[symbol] = remaining
        # Write fills to CSV + update state
        for f in fills:
            self._apply_fill_state(f)
            self._write_fill(f)

        return fills

    # ---------- Internals ---------- #
    def _simulate_order(self, ts: datetime, ord: Order, o: float, h: float, l: float, c: float) -> Tuple[bool,bool,List[Fill]]:
        """
        Return: (filled: bool, partial: bool, fills: List[Fill])
        """
        def mk_fill(qty: float, price: float) -> Fill:
            slip = price * (self.exec.slippage_bps / 10_000.0)
            px = price + slip if ord.side in (Side.BUY, Side.BUY_TO_COVER) else price - slip
            comm = abs(qty) * self.exec.fee_per_share
            return Fill(
                ts=ts, symbol=ord.symbol, side=ord.side, qty=qty, fill_price=float(px),
                commission=comm, client_order_id=ord.client_order_id,
                edge=ord.edge, edge_id=ord.edge_id, category=ord.category, meta=ord.meta
            )

        qty = float(ord.qty)
        fills: List[Fill] = []

        # MARKET: fill at open (backtest convention); IOC/FOK same
        if ord.type == OrderType.MARKET:
            fills.append(mk_fill(qty, o))
            return True, False, fills

        # LIMIT
        if ord.type == OrderType.LIMIT and ord.limit_price is not None:
            lp = float(ord.limit_price)
            # Buy if market traded at/under limit; sell if at/above
            if ord.side in (Side.BUY, Side.BUY_TO_COVER):
                if l <= lp:  # touched or crossed
                    px = min(lp, o) if o <= lp else lp
                    fills.append(mk_fill(qty, px))
                    return True, False, fills
            else:
                if h >= lp:
                    px = max(lp, o) if o >= lp else lp
                    fills.append(mk_fill(qty, px))
                    return True, False, fills
            # IOC/FOK not filled → cancel
            if ord.tif in (TimeInForce.IOC, TimeInForce.FOK):
                return False, False, []
            return False, False, []

        # STOP (stop-market)
        if ord.type == OrderType.STOP and ord.stop_price is not None:
            sp = float(ord.stop_price)
            triggered = (h >= sp) if ord.side in (Side.BUY, Side.BUY_TO_COVER) else (l <= sp)
            if triggered:
                # Market order at next available price (use open for bar trigger)
                px = o if ((ord.side in (Side.BUY, Side.BUY_TO_COVER) and o >= sp)
                           or (ord.side in (Side.SELL, Side.SELL_SHORT) and o <= sp)) else sp
                fills.append(mk_fill(qty, px))
                return True, False, fills
            if ord.tif in (TimeInForce.IOC, TimeInForce.FOK):
                return False, False, []
            return False, False, []

        # STOP-LIMIT
        if ord.type == OrderType.STOP_LIMIT and ord.stop_price is not None and ord.limit_price is not None:
            sp, lp = float(ord.stop_price), float(ord.limit_price)
            trigger = (h >= sp) if ord.side in (Side.BUY, Side.BUY_TO_COVER) else (l <= sp)
            if trigger:
                # After trigger, act like limit
                if ord.side in (Side.BUY, Side.BUY_TO_COVER):
                    if l <= lp:
                        px = min(lp, o) if o <= lp else lp
                        fills.append(mk_fill(qty, px))
                        return True, False, fills
                else:
                    if h >= lp:
                        px = max(lp, o) if o >= lp else lp
                        fills.append(mk_fill(qty, px))
                        return True, False, fills
            if ord.tif in (TimeInForce.IOC, TimeInForce.FOK):
                return False, False, []
            return False, False, []

        # Unknown or unfilled
        if ord.tif in (TimeInForce.IOC, TimeInForce.FOK):
            return False, False, []
        return False, False, []

    def _apply_fill_state(self, f: Fill) -> None:
        qty = f.qty if f.side in (Side.BUY, Side.BUY_TO_COVER) else -f.qty
        prev = self.state.positions.get(f.symbol, 0.0)
        new = prev + qty
        self.state.positions[f.symbol] = new
        # (Avg price handling omitted for brevity; integrate with your PnL engine)

    def _write_fill(self, f: Fill) -> None:
        row = [
            f.ts.strftime("%Y-%m-%d"),
            f.symbol,
            f.side.name.lower().replace("_", " "),
            float(f.qty),
            float(f.fill_price),
            float(f.commission),
            "",  # pnl computed elsewhere
            f.edge or "",
            "",  # edge_group (unused here)
            "",  # trigger
            str(f.meta or {}),
            f.edge_id or "",
            f.category or "",
        ]
        with self.trades_path.open("a", newline="") as fh:
            csv.writer(fh).writerow(row)

# portfolio/sleeves.py
from __future__ import annotations
from dataclasses import dataclass
from typing import Dict, List
import pandas as pd
import numpy as np


@dataclass
class SleeveSpec:
    name: str
    target_weight: float        # fraction of total capital
    max_gross: float = 1.0      # leverage at sleeve level
    max_pos_pct: float = 0.05   # per-name cap of sleeve NAV
    sector_cap_pct: float = 0.25  # optional sector cap


class Sleeves:
    """
    Combines edge signals into sleeve targets, and sleeve targets into portfolio targets.
    """
    def __init__(self, specs: List[SleeveSpec]) -> None:
        self.specs = {s.name: s for s in specs}

    def combine_edge_weights(self, edge_signals: Dict[str, Dict[str, float]], sleeve_name: str) -> pd.Series:
        """
        edge_signals: mapping edge_id -> {symbol -> signed weight}
        Returns a normalized per-symbol series (sum |w| <= 1) for the sleeve
        """
        if not edge_signals:
            return pd.Series(dtype=float)

        df = pd.DataFrame(edge_signals).fillna(0.0)  # rows=symbols, cols=edges
        # simple equal-weight across edges in the sleeve, then L1-norm
        sym_w = df.mean(axis=1)
        norm = sym_w.abs().sum()
        if norm > 0:
            sym_w = sym_w / norm
        return sym_w

    def apply_constraints(self, sym_w: pd.Series, sleeve: SleeveSpec) -> pd.Series:
        # clip individual names
        sym_w = sym_w.clip(lower=-sleeve.max_pos_pct, upper=sleeve.max_pos_pct)
        # re-normalize L1 to max_gross
        gross = sym_w.abs().sum()
        if gross > 0:
            sym_w = sym_w * (sleeve.max_gross / gross)
        return sym_w

    def portfolio_targets(self, sleeve_to_edges: Dict[str, Dict[str, Dict[str, float]]]) -> pd.Series:
        """
        sleeve_to_edges: sleeve -> edge_id -> {symbol -> signed weight}
        returns per-symbol total target weights (sum |w| <= sum sleeve.max_gross * sleeve.target_weight)
        """
        total = pd.Series(dtype=float)
        for sname, edges in sleeve_to_edges.items():
            spec = self.specs[sname]
            w = self.combine_edge_weights(edges, sname)
            w = self.apply_constraints(w, spec)
            total = total.add(spec.target_weight * w, fill_value=0.0)
        return total

🧱 2. Core Structural Changes (What We’re Missing and Why It Matters)

Let’s break down the 5 most critical updates we discussed and how they’ll actually reshape how the system functions.

⸻

(1) Portfolio Engine Implementation

🔍 What changes:
We move from isolated trades to a coherent portfolio model — everything lives in an “account” that tracks capital allocations, exposures, and drift.

🧠 Why it matters:
	•	Currently, each edge just fires trades independently.
	•	With a real portfolio engine:
	•	Edges compete for capital based on weights from the governor.
	•	Rebalancing ensures long-term holdings (e.g. SPY) and short-term trades (RSI mean reversion) coexist.
	•	Positions persist beyond single trades, enabling multi-day and multi-edge interactions.

⚙️ What it introduces:
	•	Target weights per edge or “sleeve.”
	•	Cash, equity, and margin management.
	•	Drift and rebalance logic.
	•	Portfolio-level risk stats (exposure, beta, VaR).

🌍 System impact:
This turns the backtester into a full Portfolio Simulator, not just a trade repeater.
It’s the key to behaving like Schwab or a real PM.

⸻

(2) Unified OMS (Order Management System)

🔍 What changes:
ExecutionSimulator (for backtests) and AlpacaBroker (for live/paper) both follow the same Order → Fill → Position model.

🧠 Why it matters:
Right now, backtest and live are two separate worlds.
A unified OMS:
	•	Lets you switch between “backtest” and “paper” modes with no code changes.
	•	Ensures trades are logged identically, so the dashboard, metrics, and governor don’t care what mode you’re in.
	•	Makes future broker integrations trivial (just subclass BrokerInterface).

⚙️ What it introduces:
	•	A shared datamodel (Order, Fill, Position).
	•	A simple event bus or message queue so all fills go through the same logger.
	•	Realistic timestamps and latency simulation (for later).

🌍 System impact:
This is what lets us run the same code live that we use for research — the single biggest jump toward production-grade infrastructure.

⸻

(3) Governor Evolution: From Passive Stats to Active Allocator

🔍 What changes:
The governor stops being just a metric recorder and becomes a capital allocator — determining what % of the portfolio goes to each edge based on performance.

🧠 Why it matters:
Currently, weights are saved but not used.
In the next version, these weights will actually control how much capital each edge gets through the PortfolioEngine.
That means:
	•	Strong edges get more weight.
	•	Weak ones fade out automatically.
	•	The system’s allocation shifts over time based on measurable performance.

⚙️ What it introduces:
	•	Per-edge metrics (Sharpe, MDD, correlation).
	•	Allocation policy (e.g., SR-weighted softmax).
	•	Min/max caps and decay.
	•	Optional diversification penalty to avoid crowding correlated edges.

🌍 System impact:
This is the start of autonomous learning — the system starts making portfolio-level decisions instead of just measuring them.

⸻

(4) Risk Engine 2.0

🔍 What changes:
The Risk Engine stops being just “order sizing” and starts managing portfolio risk holistically — similar to an internal risk desk.

🧠 Why it matters:
	•	Prevents runaway exposure when several edges align in one direction.
	•	Enforces daily loss limits and volatility targeting.
	•	Keeps portfolio within risk policy thresholds (max beta, turnover, leverage, etc.).

⚙️ What it introduces:
	•	Dynamic position limits based on vol and equity.
	•	Portfolio exposure monitoring.
	•	Real-time position re-scaling (vol-targeting logic).
	•	“Emergency brake” logic (halts trading if drawdown exceeds threshold).

🌍 System impact:
This is the part that makes the machine safe and institutionally credible — risk constraints are what separate toy algos from funds.

⸻

(5) Data Foundation and Observability

🔍 What changes:
All trade, snapshot, and governor data move to a structured database (DuckDB or SQLite) with schema validation and run IDs.

🧠 Why it matters:
Right now, our CSVs are fragile (tokenizing errors, missing fields, etc.).
A structured backend gives:
	•	Atomic writes (no partial corruption).
	•	Session-level versioning (so you can compare runs).
	•	Instant data access for dashboards, governor, or analysis scripts.

⚙️ What it introduces:
	•	run_registry table — each backtest/paper/live session is versioned.
	•	trades, snapshots, weights tables with strict schemas.
	•	Migration utilities to import/export CSVs for compatibility.

🌍 System impact:
This is the backbone that makes the machine robust, auditable, and reproducible — like a quant research platform rather than a script.
